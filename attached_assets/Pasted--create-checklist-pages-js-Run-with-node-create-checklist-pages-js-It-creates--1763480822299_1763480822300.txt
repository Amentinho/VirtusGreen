function ensureDir(dirPath) {
  if (!fs.existsSync(dirPath)) {
    fs.mkdirSync(dirPath, { recursive: true });
  }
}

function makeHreflangs(currentLang) {
  // Output hreflang links for all three languages
  return Object.keys(pages)
    .map((k) => {
      const p = pages[k];
      // In-site relative urls
      return `<link rel="alternate" hreflang="${p.lang}" href="${p.urlPath}">`;
    })
    .join("\n    ");
}

function escapeHtml(str) {
  return String(str)
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;");
}

function writePages() {
  ensureDir(OUT_DIR);
  ensureDir(ASSETS_DIR);

  Object.keys(pages).forEach((k) => {
    const dir = path.join(OUT_DIR, k);
    ensureDir(dir);
    const filePath = path.join(dir, "index.html");
    const html = generateHTML(k);
    fs.writeFileSync(filePath, html, "utf8");
    console.log("Wrote:", filePath);
  });

  console.log("\nPages created in:", OUT_DIR);
  console.log("Place your PDFs in:", ASSETS_DIR);
  console.log("Expected PDF filenames:");
  console.log(" - 7_day_sustainable_eating_checklist_en.pdf");
  console.log(" - 7_day_sustainable_eating_checklist_it.pdf");
  console.log(" - 7_day_sustainable_eating_checklist_es.pdf");
  console.log("\nEdit OG_IMAGE_PATH at top of this file to set your OG image path.\n");
}
